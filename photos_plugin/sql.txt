CREATE TABLE `photos_cats` (
  `id` int(11) NOT NULL auto_increment,
  `name` text NOT NULL,
  `img` text NOT NULL,
  `cat` int(11) NOT NULL default '0',
  PRIMARY KEY  (`id`)
)  CHARSET=utf8  AUTO_INCREMENT=1 ;


CREATE TABLE `photos_data` (
  `id` int(11) NOT NULL auto_increment,
  `name` text NOT NULL,
  `img` text NOT NULL,
  `thumb` text NOT NULL,
  `cat` int(11) NOT NULL default '0',
  `date` datetime NOT NULL default '0000-00-00 00:00:00',
  PRIMARY KEY  (`id`)
)   CHARSET=utf8 AUTO_INCREMENT=1 ;
